function Heading(props)
{
  return (<h1>Ohio State Workshop</h1>);
}

function ApiStatus(props)
{
  return (<div className="extra-left-margin">API Status: {props.text}</div>)
}

const AppComponent = React.createClass({
  
  getInitialState: function()
  {
    return
    {
      status: 'unknown'
    };
  },
  render: function() {
    return (
    <div>
      <Heading />
      <ApiStatus text={this.state.status} />
    </div>
    );
  }
});

ReactDOM.render(
    <AppComponent />,
    document.getElementById('react-root')
);




.extra-left-margin {
  margin-left: 20px;
}
